매우 어려운 문제였다... 5점 짜리 문제... 이건 좀 사고력과도 관련된 문제 같은데,

0번째 인덱스의 스티커를 뜯을 경우와 안뜯을 경우를 나누어서 2개의 DP테이블을 생성해서 풀어야 하는 문제였음..

#### python

```python
def solution(sticker):
    n = len(sticker)
    if n == 1:
        return sticker[0]

    # (1) 0번째 스티커를 뜯는 경우
    dp1 = [0] * n
    dp1[0] = dp1[1] = sticker[0] # 1번째 스티커는 뜯을 수 없으므로 0번째 값으로 초기화
    for i in range(2, n - 1): # 원형이므로 마지막 인덱스는 제외
        dp1[i] = max(sticker[i] + dp1[i - 2], dp1[i - 1])

    # 0번째 스티커를 안 뜯는 경우(1번째 스티커를 뜯는 경우)
    dp2 = [0] * n
    dp2[1] = sticker[1]
    for i in range(2, n):
        dp2[i] = max(sticker[i] + dp2[i - 2], dp2[i - 1])

    return max(max(dp1), max(dp2))
```

```js
function solution(sticker) {
  const n = sticker.length

  if (n === 1) {
    return sticker[0]
  }

  const dp1 = new Array(n).fill(0)
  dp1[0] = dp1[1] = sticker[0]
  for (let i = 2; i < n - 1; i += 1) {
    dp1[i] = Math.max(sticker[i] + dp1[i - 2], dp1[i - 1])
  }

  const dp2 = new Array(n).fill(0)
  dp2[1] = sticker[1]
  for (let i = 2; i < n; i += 1) {
    dp2[i] = Math.max(sticker[i] + dp2[i - 2], dp2[i - 1])
  }

  return Math.max(Math.max(...dp1), Math.max(...dp2))
}
```
