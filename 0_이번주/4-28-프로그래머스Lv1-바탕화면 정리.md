def solution(wallpaper):
    n = len(wallpaper)
    m = len(wallpaper[0])

    def search_row(i):
        for j in range(m):
            if wallpaper[i][j] == "#":
                return True
        return False

    def search_col(j):
        for i in range(n):
            if wallpaper[i][j] == "#":
                return True
        return False

    x1 = y1 = 0
    x2 = n - 1
    y2 = m - 1

    for i in range(n):
        if search_row(i):
            x1 = i
            break

    for i in range(n - 1, -1, -1):
        if search_row(i):
            x2 = i
            break

    for j in range(m):
        if search_col(j):
            y1 = j
            break

    for j in range(m - 1, -1, -1):
        if search_col(j):
            y2 = j
            break

    return [x1, y1, x2 + 1, y2 + 1]

def solution(wallpaper):
    n = len(wallpaper)
    m = len(wallpaper[0])

    rows = []
    cols = []
    for i in range(n):
        for j in range(m):
            if wallpaper[i][j] == "#":
                rows.append(i)
                cols.append(j)

    return [min(rows), min(cols), max(rows) + 1, max(cols) + 1]
